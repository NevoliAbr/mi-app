[1] CREAR TABLAS INICIALES
----------------------------------------------------

-- Tabla: usuarios
CREATE TABLE usuarios (
    id INT PRIMARY KEY IDENTITY(1,1),
    nombre NVARCHAR(100) NOT NULL,
    email NVARCHAR(150) NOT NULL UNIQUE,
    password_hash NVARCHAR(255) NOT NULL,
    rol NVARCHAR(20) DEFAULT 'usuario',
    activo BIT DEFAULT 1,
    creado_en DATETIME DEFAULT SYSDATETIME()
);

-- Tabla: usuario_info
CREATE TABLE usuario_info (
    id INT PRIMARY KEY IDENTITY(1,1),
    usuario_id INT NOT NULL,
    fecha_nacimiento DATE NULL,
    direccion NVARCHAR(500) NULL,
    estado_civil NVARCHAR(50) NULL,
    foto NVARCHAR(500) NULL,
    actualizado_en DATETIME DEFAULT SYSDATETIME(),
    CONSTRAINT FK_usuario_info_usuarios FOREIGN KEY (usuario_id) REFERENCES usuarios(id)
);

-- Tabla: proyectos
CREATE TABLE proyectos (
    id INT PRIMARY KEY IDENTITY(1,1),
    nombre NVARCHAR(150) NOT NULL,
    activo BIT DEFAULT 1
);

-- Tabla: usuario_proyectos (relacion N:N)
CREATE TABLE usuario_proyectos (
    usuario_id INT NOT NULL,
    proyecto_id INT NOT NULL,
    PRIMARY KEY (usuario_id, proyecto_id),
    CONSTRAINT FK_um_usuarios FOREIGN KEY (usuario_id) REFERENCES usuarios(id) ON DELETE CASCADE,
    CONSTRAINT FK_um_proyectos FOREIGN KEY (proyecto_id) REFERENCES proyectos(id) ON DELETE CASCADE
);

-- Tabla: avisos
CREATE TABLE avisos (
    id INT PRIMARY KEY IDENTITY(1,1),
    titulo NVARCHAR(200) NOT NULL,
    texto NVARCHAR(MAX) NULL,
    fecha_fin DATE NOT NULL,
    link NVARCHAR(500) NULL,
    activo BIT DEFAULT 1,
    creado_en DATETIME DEFAULT SYSDATETIME()
);

-- Tabla: aviso_imagenes
CREATE TABLE aviso_imagenes (
    id INT PRIMARY KEY IDENTITY(1,1),
    aviso_id INT NOT NULL,
    ruta NVARCHAR(500) NOT NULL,
    CONSTRAINT FK_aviso_imagenes_avisos FOREIGN KEY (aviso_id) REFERENCES avisos(id) ON DELETE CASCADE
);

-- Orden de ejecucion: usuarios → proyectos → usuario_info → usuario_proyectos → avisos → aviso_imagenes

[2] RENOMBRAR MINISTERIOS → PROYECTOS
    (Si ya creaste las tablas con el nombre anterior)
----------------------------------------------------

-- Renombrar tabla ministerios → proyectos
EXEC sp_rename 'ministerios', 'proyectos';

-- Renombrar tabla usuario_ministerios → usuario_proyectos
EXEC sp_rename 'usuario_ministerios', 'usuario_proyectos';

-- Renombrar columna ministerio_id → proyecto_id
EXEC sp_rename 'usuario_proyectos.ministerio_id', 'proyecto_id', 'COLUMN';

-- Actualizar constraint de foreign key
ALTER TABLE usuario_proyectos DROP CONSTRAINT FK_um_ministerios;
ALTER TABLE usuario_proyectos ADD CONSTRAINT FK_um_proyectos
    FOREIGN KEY (proyecto_id) REFERENCES proyectos(id) ON DELETE CASCADE;
